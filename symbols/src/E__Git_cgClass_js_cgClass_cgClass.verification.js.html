280 		 * @desc  验证单条条件
281 		 * @event
282 		 * @param {String} val 被验证数据
283 		 * @param {String} name 验证信息名称
284 		 * @param {String} ver 单条验证信息
285 		 * @param {jQuery Object} dom 被验证jQuery Object
286 		 */
287 		match : function(val, name, ver, dom){
288 			var self = this,
289 				_ver = ver.match(/^([^\,]+)\,\{?([^\}]+)\}?$/), // 匹配"ver,({)xxx(})" ==> [1]:ver, [2]:xxx
290 				term = !!_ver ? _ver[1] : ver,
291 				mapTerm = self.map[term],
292 				msg = "", doTest;
293 			// 验证条件存在时
294 			if ( !!mapTerm) {
295 				msg = mapTerm.msg && mapTerm.msg.replace("{name}", name);
296 				// 正则验证时
297 				if ( !!mapTerm.reg ) {
298 					if ( mapTerm.reg.test(val) ) {
299 						self.verified(dom);
300 						return true;
301 					}else{
302 						self.thrown(dom, msg);
303 						return false;
304 					}
305 				}
306 				// 函数验证时
307 				else if ( !!self[term] ) {
308 					if ( !!_ver ) {
309 						doTest = self[term](val, msg, dom, _ver[2]);
310 					}else{
311 						doTest = self[term](val, msg, dom, term);
312 					}
313 					if ( !!doTest.result ) {
314 						doTest.result != "ajax" && self.verified(dom);
315 					}else{
316 						self.thrown(dom, doTest.msg);
317 					}
318 					return doTest.result;
319 				}else{
320 					// 附加函数验证
321 					if ( !!term.fun ) {
322 						doTest = term.fun(val, msg, dom, term);
323 						if ( !!doTest.result ) {
324 							doTest.result != "ajax" && self.verified(dom);
325 						}else{
326 							self.thrown(dom, doTest.msg);
327 						}
328 					}
329 				}
330 			}
331 			else{
332 				alert("验证信息配置有误，请检查代码");
333 			}

468 	    		_ajax,
469 	    		_ajaxUrl = dom.attr("data-same-url"),
470 	    		_ajaxData = {

472                     id : dom.attr("data-same-id") == "" ? "" : dom.attr("data-same-id")
473                 };

475 	    		self.ajaxQueue.length++;
476 				self.ajaxQueue.isHave = {};
477 				self.ajaxQueue.isHave.length = 0;
478 			}

480 			_ajax = self.ajaxQueue.isHave.length;
481 	    	self.ajaxQueue.isHave[_ajax] = cgClass.Ajax({
482 	    		url : _ajaxUrl,

484 	    		data : _ajaxData,

499 	    			}

502 	    			self.ajaxQueue.isHave.length--;
503 	    			self.ajaxQueue.isHave[_ajax] = null;
504 	    			if ( result.success ) {
505 	    				self.verified(dom);
506 	    			} else{

508 	    				self.abortAjax();
509 	    			}
510 	    		},
511 	    		error: function( result, statusText, error ){
512 	    			self.ajaxQueue.isHave.length--;
513 	    			self.ajaxQueue.isHave[_ajax] = null;
514 	    			self.thrown(dom, "验证超时，请重新尝试");
515     				self.abortAjax();
516 	    		}
517 	    	});
