1 /**
  2  * @author 徐晨 
  3  * @name cgClass.Autocomplete
  4  * @class 自动完成插件 <a href="../demo/autocomplete.html" target="_blank">demo</a>
  5  * @constructor
  6  * @extends cgClass
  7  * @extends cgClass.Ajax
  8  * @extends jQuery
  9  * @extends jQuery.getFromTemplate
 10  * @since version 1.0 
 11  * @param {Object} options 参数对象数据
 12  * @param {jQuery Object} options.wrap 包裹的jquery dom
 13  * @param {String html} options.template 展示项的js模板
 14  * @param {jQuery class} options.autoList 展示列表的钩子名称
 15  * @param {jQuery class} options.autoItem 展示项的钩子名称
 16  * @param {jQuery class} options.autoText 输入框的钩子名称
 17  * @param {jQuery Object} options.autoBtn 提交按钮的jquery dom
 18  * @param {Boolean} options.hasBtn 是否存在提交按钮
 19  * @param {css class} options.hoverClass 悬停显示项时添加的样式名
 20  * @param {Function} options.init 模块初始化时的附加事件
 21  * @param {Function} options.createSearchList 生成显示列表的事件
 22  * @param {Function} options.action 表单提交的事件
 23  */
 24 cgClass.AddClass(
 25     "Autocomplete",
 26     {
 27         init : function(options){
 28             var self = this,
 29                 option = $.extend(/** @lends cgClass.Autocomplete.prototype*/{
 30                     /**
 31                      * 需要验证的表格包裹的jquery dom
 32                      * @type jQuery Object
 33                      */
 34                     wrap : $(".Js-auto-frame"),
 35                     /**
 36                      * 展示列表的钩子名称
 37                      * @type String html
 38                      * @default null 
 39                      */
 40                     template : null,
 41                     /**
 42                      * 展示列表的钩子名称
 43                      * @type jQuery class
 44                      * @default ".Js-auto-list"
 45                      */
 46                     autoList : ".Js-auto-list",
 47                     /**
 48                      * 展示项的钩子名称
 49                      * @type jQuery class
 50                      * @default ".Js-auto-item"
 51                      */
 52                     autoItem : ".Js-auto-item",
 53                     /**
 54                      * 输入框的钩子名称
 55                      * @type jQuery class
 56                      * @default ".Js-auto-text"
 57                      */
 58                     autoText : ".Js-auto-text",
 59                     /**
 60                      * 提交按钮的jquery dom
 61                      * @type jQuery dom
 62                      * @default $(".Js-auto-btn")
 63                      */
 64                     autoBtn : $(".Js-auto-btn"),
 65                     /**
 66                      * 是否存在提交按钮
 67                      * @type Boolean
 68                      * @default true
 69                      */
 70                     hasBtn : true,
 71                     /**
 72                      * 悬停显示项时添加的样式名
 73                      * @type css class
 74                      * @default "selected"
 75                      */
 76                     hoverClass : "selected",
 77                     /**
 78                      * 模块初始化时的附加事件
 79                      * @type Function
 80                      * @default null || init : function(option){}
 81                      */
 82                     init : function(option){},
 83                     /**
 84                      * 生成显示列表的事件
 85                      * @type Function
 86                      * @default function(option, val){}
 87                      */
 88                     createSearchList : function(option, val, addItem){},
 89                     /**
 90                      * 表单提交的事件
 91                      * @type Function
 92                      * @default function(option, val){}
 93                      */
 94                     action : function(option, val){}
 95                 }, options),
 96                 $wrap = option.wrap,
 97                 $list = $wrap.find(option.autoList),
 98                 $text = $wrap.find(option.autoText),
 99                 $btn = $wrap.find(option.autoBtn),
100                 $temp = option.template || '<li class="auto-item Js-auto-item hide-row"><a href="javascript:void(0)" title="#{title}">#{title}</a></li>',
101                 keys = {
102                     back: 8,
103                     enter:  13,
104                     escape: 27,
105                     up:     38,
106                     down:   40,
107                     array:  [13, 27, 38, 40]
108                 },
109                 keyDate = [],
110                 chooseData = [];
111 
112             self.wrap = $wrap;
113             self.list = $list;
114             self.text = $text;
115             self.btn = $btn;
116             self.itemTemp = $temp;
117 
118             // out param
119             self.outParam = self.applyMethods(self, {
120                 oWrap : $wrap,
121                 oList : $list,
122                 oText : $text,
123                 oBtn : $btn,
124                 returnText : self.returnText,
125                 addItem : self.addItem,
126                 showList : self.showList,
127                 closeList : self.closeList
128             });
129 
130 
131             // bind when auto item hover change item's class
132             $wrap.on(
133                 "mouseenter",
134                 option.autoItem,
135                 function(){
136                     $(this).addClass(option.hoverClass);
137                 }
138             ).on(
139                 "mouseleave",
140                 option.autoItem,
141                 function(){
142                     $(this).removeClass(option.hoverClass);
143                 }
144             ).on(
145             // bind when auto item clicked return function
146                 "click",
147                 option.autoItem,
148                 function(){
149                     self.closeList();
150                     if ( typeof( option.action ) == "function" ) {
151                         option.action(self.outParam, $.trim( $(this).children().attr("title") ), $(this) );
152                     }
153                     self.returnText( $(this).children().attr("title") );
154                 }
155             ).on(
156             // bind when auto item keyup return function
157                 "keyup",
158                 option.autoText,
159                 function(event){
160                     var keyCode = event.keyCode;
161                     keyDate.push((new Date()).getTime());
162                     var thisTime = keyDate[(keyDate.length - 1)];
163                     var val = "";
164                     if($.inArray(keyCode, keys.array) !=-1){
165                         val = $.trim( $text.val() ); 
166                         if ( $list.is(":hidden") && $list.children(option.autoItem).length ) {
167                             self.showList();
168                         }
169                         var $item = $list.children(option.autoItem),
170                             $cur = $item.filter("." + option.hoverClass);
171                         switch (keyCode){
172                             case keys.up:
173                                 if ( $cur.length ){
174                                     $cur.removeClass(option.hoverClass).prev().addClass(option.hoverClass);
175                                 }else{
176                                     $item.last().addClass(option.hoverClass);
177                                 }
178                             break;
179                             case keys.down:
180                                 if ( $cur.length ){
181                                     $cur.removeClass(option.hoverClass).next().addClass(option.hoverClass);
182                                 }else{
183                                     $item.first().addClass(option.hoverClass);
184                                 }
185                             break;
186                             case keys.enter:
187                                 $list.hide();
188                                 if ( typeof( option.action ) == "function" ) {
189                                     if ( $cur.length ) {
190                                         $cur.click();
191                                     } else{
192                                         option.action(self.outParam, $.trim( $text.val() ), $cur );
193                                     }
194                                 }
195                             break;
196                             default :
197                             break;
198                         }
199                     }else{
200                         // 延时显示searchlist 减少多余查询
201                         setTimeout(function(){
202                             val = $.trim( $text.val() );
203                             if( thisTime == keyDate[(keyDate.length - 1)] && val != "" ) {
204                                 option.createSearchList(self.outParam, val);
205                             }else if ( val == "" ){
206                                 self.closeList();
207                             }
208                         }, 300);
209                     }
210                 }
211             ).on(
212             // bind when auto item keydown return function
213                 "keydown",
214                 option.autoText,
215                 function(event){
216                     var keyCode = event.keyCode;
217                     if ( keyCode == keys.back && $.trim( $(this).val() ) == "" && $list.is(":visible") ){
218                         self.closeList();
219                     }
220                 }
221             ).on(
222             // bind when auto item focus return function
223                 "focus",
224                 option.autoText,
225                 function(){
226                     if( $.trim( $(this).val() ) != "" && $list.children().length && $list.is(":hidden") ){
227                         self.showList();
228                     }
229                 }
230             );
231 
232             if ( option.hasBtn ) {
233                 option.autoBtn.on("click", function(){
234                     self.closeList();
235                     if ( typeof( option.action ) == "function" ) {
236                         option.action(self.outParam, $.trim( $text.val() ) );
237                     }
238                 });
239             }
240 
241             // init
242             if ( typeof option.init == "function" ) {
243                 option.init(self.outParam);
244             }
245         },
246         /**
247          * @name cgClass.Autocomplete#returnText
248          * @desc  替换输出框的值为带入参数
249          * @event
250          * @param {String} value 替换输入框的值
251          */
252         returnText : function(value){
253             this.text.val( $.trim(value) );
254         },
255         /**
256          * @name cgClass.Autocomplete#addItem
257          * @desc  添加展示项
258          * @event
259          * @param {Array} data 展示项的数据 例:["abc", "efg"]
260          */
261         addItem : function(data){
262             var self = this;
263             self.list.empty();
264             if ( data.constructor === Array && data.length ) {
265                 $.each(data, function(index, item){
266                     self.list.append( $.getFromTemplate( self.itemTemp, item ) );
267                 });
268                 self.showList();
269             }else{
270                 self.closeList();
271             }
272         },
273         /**
274          * @name cgClass.Autocomplete#showList
275          * @desc  显示显示列表，并绑定隐藏列表功能
276          * @event
277          */
278         showList : function(){
279             var self = this;
280             self.list.slideDown("fast", "linear", function(){
281                 $(document).one("click", function(){
282                     self.list.hide();
283                 });
284             });
285         },
286         /**
287          * @name cgClass.Autocomplete#closeList
288          * @desc  隐藏显示列表
289          * @event
290          */
291         closeList : function(){
292             this.list.hide();
293         }
294     }
295 );
